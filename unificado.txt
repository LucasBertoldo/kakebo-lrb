function detectarBancoArquivo(texto, nomeArquivo) {
    const txtCompleto = (texto + " " + nomeArquivo).toUpperCase();
    for (const key in BANCOS_BRANDING) {
        if (BANCOS_BRANDING[key].keywords.some(kw => txtCompleto.includes(kw))) {
            return BANCOS_BRANDING[key];
        }
    }
    return null;
}

function obterCorTextoContraste(hexcolor) {
    if (!hexcolor || hexcolor.startsWith('var')) return 'var(--texto-principal)';
    let hex = hexcolor.replace('#', '');
    if (hex.length === 3) hex = hex[0]+hex[0]+hex[1]+hex[1]+hex[2]+hex[2];
    const r = parseInt(hex.substring(0, 2), 16);
    const g = parseInt(hex.substring(2, 4), 16);
    const b = parseInt(hex.substring(4, 6), 16);
    const yiq = ((r * 299) + (g * 587) + (b * 114)) / 1000;
    return (yiq >= 135) ? '#1e293b' : '#ffffff';
}

function detectarEProcessarArquivo(input) {
    const arq = input.files[0]; if(!arq) return; const leitor = new FileReader();
    if(arq.name.endsWith(".pdf")) {
        UI.showToast("Analisando PDF...", "success");
        leitor.onload = async function() { const doc = await pdfjsLib.getDocument({ data: new Uint8Array(this.result) }).promise; let txt = "";
        for(let p=1; p<=doc.numPages; p++) txt += (await (await doc.getPage(p)).getTextContent()).items.map(i => i.str).join(" ") + " "; processarImportacao(txt, arq.name, processarTextoRegex); }; leitor.readAsArrayBuffer(arq);
    } else if(arq.name.endsWith(".ofx")) {
        UI.showToast("Analisando OFX...", "success");
        leitor.onload = function() { processarImportacao(this.result, arq.name, processarMotorOFX); }; leitor.readAsText(arq, 'UTF-8');
    }
}

function processarImportacao(conteudo, nomeArq, funcaoExtrair) {
    let bancoConfig = detectarBancoArquivo(conteudo, nomeArq); let nomeColunaFinal = document.getElementById('imp-coluna').value;
    if (bancoConfig) { nomeColunaFinal = bancoConfig.nome.toUpperCase(); if (!colunasAtivas.includes(nomeColunaFinal)) { colunasAtivas.push(nomeColunaFinal); localStorage.setItem('k_colunas', JSON.stringify(colunasAtivas)); UI.showToast(`Conta autodetectada!`);
    } }
    const adicoes = funcaoExtrair(conteudo, nomeColunaFinal);
    if(adicoes > 0) { localStorage.setItem('k_lancamentos', JSON.stringify(lancamentos)); UI.showToast(`${adicoes} transações processadas.`); fecharModalImportador();
    renderizarPainelGeral(); } else { UI.showToast("Nenhuma transação válida lida.", "error"); }
}

function processarMotorOFX(texto, colDestino) {
    let add = 0;
    const blocosTransacionais = texto.split(/<STMTTRN>/i);
    for (let i = 1; i < blocosTransacionais.length; i++) {
        const bloco = blocosTransacionais[i];
        const valorMatch = bloco.match(/<TRNAMT>([\d.-]+)/i); let memoMatch = bloco.match(/<MEMO>([^<]+)/i) || bloco.match(/<NAME>([^<]+)/i);
        if (valorMatch && memoMatch) {
            let val = parseFloat(valorMatch[1]);
            let desc = memoMatch[1].trim(); if (isNaN(val) || val === 0) continue; let tipo = "SAÍDA";
            if (val < 0) { tipo = "SAÍDA"; val = Math.abs(val); } else { tipo = "ENTRADA"; }
            let cat = "Outros";
            for (const [k, p] of Object.entries(DICIONARIO_CATEGORIAS)) if (p.some(pal => desc.toLowerCase().includes(pal))) cat = k;
            lancamentos.push({ id: Date.now() + Math.random(), desc: UI.escapeHTML(desc), nota: "", valor: val, tipo, coluna: tipo === "ENTRADA" ? "ENTRADAS" : colDestino, categorias: [cat], membro: membrosFamilia[0], mes: mesCorrente });
            add++;
        }
    } return add;
}

function processarTextoRegex(texto, colDestino) {
    let add = 0;
    const regex = /(\d{2}\/\d{2})\s+((?:(?!\d{2}\/\d{2}).)+?)(?:R\$\s*)?(-?\d{1,3}(?:\.\d{3})*,\d{2})/g; let m;
    while((m = regex.exec(texto.replace(/\s+/g, ' '))) !== null) {
        let desc = m[2].trim();
        let val = parseFloat(m[3].replace(/\./g, '').replace(',', '.')); if(isNaN(val) || val === 0 || desc.length < 3) continue;
        let tipo = val < 0 ? "ENTRADA" : "SAÍDA"; val = Math.abs(val); let cat = "Outros";
        for (const [k, p] of Object.entries(DICIONARIO_CATEGORIAS)) if (p.some(pal => desc.toLowerCase().includes(pal))) cat = k;
        lancamentos.push({ id: Date.now()+Math.random(), desc: UI.escapeHTML(desc), nota: "", valor: val, tipo, coluna: tipo==="ENTRADA"?"ENTRADAS":colDestino, categorias: [cat], membro: membrosFamilia[0], mes: mesCorrente }); add++;
    } return add;
}

    if(!valoresVisiveis) return "R$ •••••";

}

function processarGraficos(dados) {
    const mapa = {}; let tot = 0;
    dados.filter(l => l.tipo === "SAÍDA").forEach(i => {
        const tags = Array.isArray(i.categorias) ? i.categorias : [i.categorias || "Outros"];
        if (tags.some(c => filtrosCategoriasAtivos.includes(c))) tot += i.valor;
        tags.forEach(c => { if (filtrosCategoriasAtivos.includes(c)) mapa[c] = (mapa[c] || 0) + i.valor; });
    });
    document.getElementById('lbl-total-grafico').innerText = formataValorVisivel(tot);

    if(chartPizzaInstance) chartPizzaInstance.destroy();
    
    const corTextoLabel = temaAtual === 'LIGHT' ? '#1e293b' : '#f8fafc';

    chartPizzaInstance = new Chart(document.getElementById('meuGraficoPizza').getContext('2d'), { 
        type: 'pie', 
        data: { 
            labels: Object.keys(mapa), 
            datasets: [{ 
                data: Object.values(mapa), 
                backgroundColor: coresGraficos[temaAtual], 
                borderWidth: 2, borderColor: temaAtual === 'LIGHT' ? '#ffffff' : '#1e293b'
            }] 
        }, 
        options: { 
            responsive: true, 
            maintainAspectRatio: false, 
            layout: { padding: { left: 40, right: 40, top: 40, bottom: 40 } },
            plugins: { 
                legend: { display: false }, 
                datalabels: { 
                    display: true,
                    color: corTextoLabel,
                    font: { weight: '800', size: 10 },
                    textAlign: 'center',
                    anchor: 'end',
                    align: 'end',
                    offset: 8,
                    clip: false,
                    formatter: (value, ctx) => {
                        let label = ctx.chart.data.labels[ctx.dataIndex];
                        let percentage = tot > 0 ? ((value / tot) * 100).toFixed(1) + '%' : '0%';
                        let vTxt = valoresVisiveis ? `R$ ${value.toLocaleString('pt-BR', {minimumFractionDigits: 2})}` : `R$ •••••`;
                        return `${label}\n${vTxt}\n(${percentage})`;
                    }
                } 
            } 
        } 
    });

    const linha = ["2026-04", "2026-05", "2026-06"].map(m => lancamentos.reduce((a, l) => (l.mes===m && l.tipo==="SAÍDA") ? a+l.valor : a, 0));
